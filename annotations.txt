// No Typescript existe a possibilidade de escolher as propriedades
// de um tipo ao herdá-lo.

// Exemplo com base na tipagem de um 'Styled-component':
export const Title = styled.h1<
  Pick<HeadingProps, 'colorDark' | 'size' | 'uppercase'>
>`
  ${({ theme, colorDark, size, uppercase }) => css`
    color: ${colorDark ? theme.colors.primaryColor : theme.colors.white};
    ${titleSize[size](theme)};
    ${titleCase(uppercase)};
  `}
`;
// Assim, são evitados erros relacionados ao desuso de alguma propriedade
// oriunda do tipo.
